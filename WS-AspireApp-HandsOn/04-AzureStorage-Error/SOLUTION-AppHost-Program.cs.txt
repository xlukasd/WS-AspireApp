var azureStorage = builder.AddAzureStorage("storage").ConfigureInfrastructure(infra => {
    var storageAccount = infra.GetProvisionableResources().OfType<StorageAccount>().Single();
    storageAccount.AllowSharedKeyAccess = true;
});

var azureStorageBlobs = azureStorage.AddBlobs("blobs");

var apiService = builder.AddProject<Projects.WS_AspireApp_ApiService>("apiservice")
    .WithReference(azureStorageBlobs)
    .WaitFor(azureStorageBlobs);
